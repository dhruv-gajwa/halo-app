    ```
    export function Step1AccountPage(): React.JSX.Element {
      const navigate = useNavigate()
      const draft = readWizardDraft().step1 ?? {}
      const form = useForm<Step1Values>({
        resolver: zodResolver(step1Schema),
        defaultValues: { email: '', password: '', firstName: '', lastName: '', username: '', ...draft },
        mode: 'onSubmit',  // matches UI-SPEC: validation fires on Next click, not on every keystroke
      })

      const onSubmit = form.handleSubmit((values) => {
        // Uniqueness checks AFTER schema validation has passed.
        const emailDup = findVisitorByEmail(values.email)
        const usernameDup = findVisitorByUsername(values.username)
        if (emailDup) {
          form.setError('email', {
            type: 'manual',
            // We embed JSX through the field's error message â€” Mantine renders ReactNode in `error=` slot
            message: 'An account with this email already exists.',
          })
          // Note: the inline "Sign in instead?" anchor cannot be embedded in a string message;
          // we render it conditionally below via a separate slot if the field error is the duplicate-email case.
          return
        }
        if (usernameDup) {
          form.setError('username', { type: 'manual', message: 'That username is taken â€” try another.' })
          return
        }
        writeWizardDraftStep('step1', values)
        navigate('/signup/details')
      })

      const emailErr = form.formState.errors.email?.message
      const isEmailDuplicate = emailErr === 'An account with this email already exists.'

      return (
        <Stack gap="md">
          <Title order={2}>Create your Halo account</Title>
          <Text c="dimmed">Welcome to Halo. Tell us who you are and we'll get your workspace ready.</Text>
          <form onSubmit={onSubmit} noValidate>
            <Stack gap="md">
              <TextInput
                {...form.register('email')}
                label="Email"
                placeholder="you@example.com"
                description="We'll use this for sign-in. No real emails ever leave your browser."
                error={
                  isEmailDuplicate ? (
                    <span>
                      An account with this email already exists.{' '}
                      <Anchor href="/signin" pendoId={PENDO_IDS.signup.step1.signinAnchor}>Sign in instead?</Anchor>
                    </span>
                  ) : emailErr
                }
                pendoId={PENDO_IDS.signup.step1.email}
              />
              <PasswordInput
                {...form.register('password')}
                label="Password"
                placeholder="At least 8 characters"
                description="Hashed locally and stored in your browser â€” there is no server."
                error={form.formState.errors.password?.message}
                pendoId={PENDO_IDS.signup.step1.password}
              />
              <TextInput
                {...form.register('firstName')}
                label="First name"
                placeholder="Ada"
                error={form.formState.errors.firstName?.message}
                pendoId={PENDO_IDS.signup.step1.firstName}
              />
              <TextInput
                {...form.register('lastName')}
                label="Last name"
                placeholder="Lovelace"
                error={form.formState.errors.lastName?.message}
                pendoId={PENDO_IDS.signup.step1.lastName}
              />
              <TextInput
                {...form.register('username')}
                label="Username"
                placeholder="ada"
                description="Visible on your profile and team page."
                error={form.formState.errors.username?.message}
                pendoId={PENDO_IDS.signup.step1.username}
              />
              <Group justify="flex-end" mt="xl">
                {/* Step 1 has NO Back button per UI-SPEC "Page CTAs" table */}
                <Button type="submit" loading={form.formState.isSubmitting} pendoId={PENDO_IDS.signup.step1.submit}>
                  Continue
                </Button>
              </Group>
            </Stack>
          </form>
        </Stack>
      )
    }
    ```
