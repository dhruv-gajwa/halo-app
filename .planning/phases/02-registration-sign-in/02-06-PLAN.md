    export function RequireAnon({ children }: { children?: ReactNode }): React.JSX.Element {
      const isAuthenticated = useAuthStore((s) => s.isAuthenticated)

        return <Navigate to="/app" replace />
      }
      return <>{children ?? <Outlet />}</>
    }
